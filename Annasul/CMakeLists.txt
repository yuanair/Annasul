    set(Annasul_SOURCES ${Annasul_SOURCES} ${CPP_FILES} PARENT_SCOPE)

function(traverse_directories_and_add_files dir)
    file(GLOB SUBDIRS "${dir}/*")
    foreach(SUBDIR ${SUBDIRS})
        if(IS_DIRECTORY ${SUBDIR})
            get_filename_component(SUBDIR_NAME ${SUBDIR} REALPATH)
            message(STATUS "Subdirectory: ${SUBDIR_NAME}")
            if(SUBDIR_NAME STREQUAL "Windows")
                if (CMAKE_SYSTEM_NAME STREQUAL "Windows")
                    traverse_and_add_files(${Annasul_SOURCES} ${Annasul_HEADERS} ${SUBDIR})
                endif()
            elseif (SUBDIR_NAME STREQUAL "Linux")
                if (CMAKE_SYSTEM_NAME STREQUAL "Linux")
                    traverse_and_add_files(${Annasul_SOURCES} ${Annasul_HEADERS} ${SUBDIR})
                endif()
            elseif (SUBDIR_NAME STREQUAL "Mac")
                if (CMAKE_SYSTEM_NAME STREQUAL "Darwin")
                    traverse_and_add_files(${Annasul_SOURCES} ${Annasul_HEADERS} ${SUBDIR})
                endif()
            elseif (SUBDIR_NAME STREQUAL "Android")
                if (CMAKE_SYSTEM_NAME STREQUAL "Android")
                    traverse_and_add_files(${Annasul_SOURCES} ${Annasul_HEADERS} ${SUBDIR})
                endif()
            else()
                traverse_and_add_files(${Annasul_SOURCES} ${Annasul_HEADERS} ${SUBDIR})
            endif ()
        endif()
    endforeach()
endfunction()


function(add_annasul_module path depend_libs)
    set(Annasul_SOURCES "")
    set(Annasul_HEADERS "")
    set(Annasul_INCLUDE_DIRS "")

    get_filename_component(SUBDIR_NAME ${path} NAME)
    message(STATUS "Module: ${SUBDIR_NAME}")

    traverse_and_add_files(${path}/Private)
    traverse_directories_and_add_files(${path}/Private)
    traverse_and_add_files(${path}/Public)
    traverse_directories_and_add_files(${path}/Public)
    set(Annasul_INCLUDE_DIRS ${Annasul_INCLUDE_DIRS} ${path}/Public)

    add_library(Annasul${SUBDIR_NAME} STATIC ${Annasul_SOURCES} ${Annasul_HEADERS})
    target_include_directories(Annasul${SUBDIR_NAME} PUBLIC ${Annasul_INCLUDE_DIRS})

    if (depend_libs)
        message(STATUS "Depend: ${depend_libs}")
        target_link_libraries(Annasul${SUBDIR_NAME} PUBLIC ${depend_libs})
    endif ()
endfunction()

add_annasul_module("./Source/Runtime/Core" "")
